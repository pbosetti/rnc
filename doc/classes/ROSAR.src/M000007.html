  def data_frame(name, df)
    Thread.new do
      @r.cmd "#{name}<-read.table('#{FIFO}', h=T)"
    end

    File.open("#{FIFO}", "w") do |f|
      f.puts df.keys*"\t"
      df[df.keys[0]].size.times do |i|
        df.each_key do |k|
          f.print "#{df[k][i]}\t"
        end
        f.puts
      end
    end
  end
